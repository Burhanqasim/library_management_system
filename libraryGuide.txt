>nest generate module member
>nest generate module book
>nest generate module author

>nest generate controller member --no-spec

>nest generate service book/providers/book --flat --no-spec
>nest generate service author/providers/author --flat --no-spec
>nest generate service borrowing/providers/borrowing --flat --no-spec
>nest generate service membership-card/providers/membership-card --flat --no-spec
